## 🎯 Introducción

**RBAC (Role-Based Access Control)** es el modelo de seguridad que implementaremos para el sistema EZ ORDER. Este documento guía la implementación paso a paso, permitiendo migrar gradualmente desde el sistema actual.

### Beneficios de RBAC:
- ✅ **Seguridad Granular**: Cada acción protegida por permisos específicos
- ✅ **Escalabilidad**: Fácil agregar nuevos roles y permisos
- ✅ **Mantenibilidad**: Políticas centralizadas y fáciles de auditar
- ✅ **Multi-tenant**: Usuarios solo acceden a su restaurante asignado

## 🔐 Fase 2: Auth Hook

### 2.1 Crear Función del Auth Hook

```sql
-- Crear función para el Auth Hook
CREATE OR REPLACE FUNCTION public.custom_access_token_hook(event JSONB)
RETURNS JSONB
LANGUAGE plpgsql
STABLE
AS $$
DECLARE
    claims JSONB;
    user_role public.app_role;
    user_restaurante_id UUID;
BEGIN
    -- Obtener el rol del usuario
    SELECT ur.role, ur.restaurante_id 
    INTO user_role, user_restaurante_id
    FROM public.user_roles ur
    WHERE ur.user_id = (event->>'user_id')::UUID
    LIMIT 1;

    claims := event->'claims';

    -- Agregar claims personalizados
    IF user_role IS NOT NULL THEN
        claims := jsonb_set(claims, '{user_role}', to_jsonb(user_role::text));
        claims := jsonb_set(claims, '{restaurante_id}', to_jsonb(user_restaurante_id));
    ELSE
        claims := jsonb_set(claims, '{user_role}', 'null');
        claims := jsonb_set(claims, '{restaurante_id}', 'null');
    END IF;

    -- Actualizar el evento con los nuevos claims
    event := jsonb_set(event, '{claims}', claims);

    RETURN event;
END;
$$;
```

### 2.2 Configurar Permisos del Auth Hook

```sql

GRANT EXECUTE ON FUNCTION public.custom_access_token_hook TO supabase_auth_admin;
```

## ⚡ Fase 3: Funciones de Autorización

### 3.1 Función Principal de Autorización

```sql
-- Función para verificar permisos
CREATE OR REPLACE FUNCTION public.authorize(
    requested_permission public.app_permission
)
RETURNS BOOLEAN AS $$
DECLARE
    bind_permissions INT;
    user_role public.app_role;
BEGIN
    -- Obtener el rol del usuario desde el JWT
    SELECT (auth.jwt() ->> 'user_role')::public.app_role INTO user_role;

    -- Verificar si el rol tiene el permiso
    SELECT COUNT(*)
    INTO bind_permissions
    FROM public.role_permissions
    WHERE role_permissions.permission = requested_permission
        AND role_permissions.role = user_role;

    RETURN bind_permissions > 0;
END;
$$ LANGUAGE plpgsql STABLE SECURITY DEFINER SET search_path = '';
```

### 3.2 Función de Autorización por Restaurante

```sql
-- Función para verificar acceso a restaurante específico
CREATE OR REPLACE FUNCTION public.authorize_restaurante(
    target_restaurante_id UUID
)
RETURNS BOOLEAN AS $$
DECLARE
    user_role public.app_role;
    user_restaurante_id UUID;
BEGIN
    -- Obtener datos del usuario desde el JWT
    SELECT 
        (auth.jwt() ->> 'user_role')::public.app_role,
        (auth.jwt() ->> 'restaurante_id')::UUID
    INTO user_role, user_restaurante_id;

    -- Super admin puede acceder a cualquier restaurante
    IF user_role = 'super_admin' THEN
        RETURN TRUE;
    END IF;

    -- Otros roles solo pueden acceder a su restaurante asignado
    RETURN user_restaurante_id = target_restaurante_id;
END;
$$ LANGUAGE plpgsql STABLE SECURITY DEFINER SET search_path = '';
```

### 3.3 Funciones de Utilidad

```sql
-- Función para obtener permisos del usuario actual
CREATE OR REPLACE FUNCTION public.get_user_permissions()
RETURNS TABLE(permission public.app_permission) AS $$
BEGIN
    RETURN QUERY
    SELECT rp.permission
    FROM public.role_permissions rp
    WHERE rp.role = (auth.jwt() ->> 'user_role')::public.app_role;
END;
$$ LANGUAGE plpgsql STABLE SECURITY DEFINER;

-- Función para obtener información del usuario actual
CREATE OR REPLACE FUNCTION public.get_current_user_info()
RETURNS JSONB AS $$
DECLARE
    result JSONB;
BEGIN
    SELECT jsonb_build_object(
        'user_id', auth.uid(),
        'role', auth.jwt() ->> 'user_role',
        'restaurante_id', auth.jwt() ->> 'restaurante_id',
        'permissions', (
            SELECT jsonb_agg(rp.permission)
            FROM public.role_permissions rp
            WHERE rp.role = (auth.jwt() ->> 'user_role')::public.app_role
        )
    ) INTO result;
    
    RETURN result;
END;
$$ LANGUAGE plpgsql STABLE SECURITY DEFINER;

-- Configurar permisos para estas funciones
GRANT EXECUTE ON FUNCTION public.get_user_permissions() TO authenticated;
GRANT EXECUTE ON FUNCTION public.get_current_user_info() TO authenticated;
```

## 🎨 Fase 5: Frontend - Composables

### 5.1 Crear Composable useRBAC

Crear archivo: `src/composables/useRBAC.ts`

```typescript
import { ref, computed } from 'vue';
import { useAuthStore } from '@/stores/auth_store';
import { supabase } from '@/plugins/supabase';

export function useRBAC() {
  const authStore = useAuthStore();
  const userPermissions = ref<string[]>([]);
  const userRole = ref<string>('');
  const userRestauranteId = ref<string>('');

  // Obtener información del usuario con permisos
  const fetchUserInfo = async () => {
    try {
      const { data, error } = await supabase.rpc('get_current_user_info');
      if (error) throw error;
      
      userRole.value = data.role || '';
      userRestauranteId.value = data.restaurante_id || '';
      userPermissions.value = data.permissions || [];
    } catch (error) {
      console.error('Error fetching user info:', error);
    }
  };

  // Verificar si el usuario tiene un permiso específico
  const hasPermission = (permission: string): boolean => {
    return userPermissions.value.includes(permission);
  };

  // Verificar si el usuario tiene alguno de los permisos
  const hasAnyPermission = (permissions: string[]): boolean => {
    return permissions.some(permission => userPermissions.value.includes(permission));
  };

  // Verificar si el usuario tiene todos los permisos
  const hasAllPermissions = (permissions: string[]): boolean => {
    return permissions.every(permission => userPermissions.value.includes(permission));
  };

  // Verificar si el usuario es un rol específico
  const hasRole = (role: string): boolean => {
    return userRole.value === role;
  };

  // Verificar si el usuario tiene alguno de los roles
  const hasAnyRole = (roles: string[]): boolean => {
    return roles.includes(userRole.value);
  };

  // Computed para mostrar información del rol
  const roleDisplayName = computed(() => {
    const roleNames: Record<string, string> = {
      'super_admin': 'Super Administrador',
      'restaurante_admin': 'Administrador',
      'mesero': 'Mesero',
      'cocina': 'Cocina',
      'caja': 'Caja',
      'inventario': 'Inventario',
      'cliente': 'Cliente'
    };
    return roleNames[userRole.value] || 'Usuario';
  });

  return {
    userPermissions,
    userRole,
    userRestauranteId,
    roleDisplayName,
    fetchUserInfo,
    hasPermission,
    hasAnyPermission,
    hasAllPermissions,
    hasRole,
    hasAnyRole
  };
}
```

### 5.2 Crear Composable usePermissions

Crear archivo: `src/composables/usePermissions.ts`

```typescript
import { computed } from 'vue';
import { useRBAC } from './useRBAC';

export function usePermissions() {
  const { hasPermission, hasAnyPermission, hasRole, hasAnyRole } = useRBAC();

  // Permisos específicos para pedidos
  const canCreatePedidos = computed(() => hasPermission('pedidos.create'));
  const canReadPedidos = computed(() => hasPermission('pedidos.read'));
  const canUpdatePedidos = computed(() => hasPermission('pedidos.update'));
  const canDeletePedidos = computed(() => hasPermission('pedidos.delete'));
  const canCancelPedidos = computed(() => hasPermission('pedidos.cancel'));

  // Permisos específicos para menú
  const canManageMenu = computed(() => hasAnyPermission(['menu.create', 'menu.update', 'menu.delete']));
  const canReadMenu = computed(() => hasPermission('menu.read'));

  // Permisos específicos para inventario
  const canManageInventario = computed(() => hasAnyPermission(['inventario.create', 'inventario.update', 'inventario.delete']));
  const canReadInventario = computed(() => hasPermission('inventario.read'));

  // Permisos específicos para clientes
  const canManageClientes = computed(() => hasAnyPermission(['clientes.create', 'clientes.update', 'clientes.delete']));
  const canReadClientes = computed(() => hasPermission('clientes.read'));

  // Permisos específicos para usuarios
  const canManageUsers = computed(() => hasAnyPermission(['usuarios.create', 'usuarios.update', 'usuarios.delete']));
  const canReadUsers = computed(() => hasPermission('usuarios.read'));

  // Permisos para reportes
  const canViewReports = computed(() => hasPermission('reportes.read'));

  // Permisos para configuración
  const canUpdateConfig = computed(() => hasPermission('config.update'));

  // Roles específicos
  const isSuperAdmin = computed(() => hasRole('super_admin'));
  const isRestauranteAdmin = computed(() => hasRole('restaurante_admin'));
  const isMesero = computed(() => hasRole('mesero'));
  const isCocina = computed(() => hasRole('cocina'));
  const isCaja = computed(() => hasRole('caja'));
  const isInventario = computed(() => hasRole('inventario'));
  const isCliente = computed(() => hasRole('cliente'));

  // Roles administrativos
  const isAdmin = computed(() => hasAnyRole(['super_admin', 'restaurante_admin']));

  return {
    // Permisos de pedidos
    canCreatePedidos,
    canReadPedidos,
    canUpdatePedidos,
    canDeletePedidos,
    canCancelPedidos,
    
    // Permisos de menú
    canManageMenu,
    canReadMenu,
    
    // Permisos de inventario
    canManageInventario,
    canReadInventario,
    
    // Permisos de clientes
    canManageClientes,
    canReadClientes,
    
    // Permisos de usuarios
    canManageUsers,
    canReadUsers,
    
    // Permisos generales
    canViewReports,
    canUpdateConfig,
    
    // Roles
    isSuperAdmin,
    isRestauranteAdmin,
    isMesero,
    isCocina,
    isCaja,
    isInventario,
    isCliente,
    isAdmin
  };
}
```

```vue
<template>
  <div v-if="hasAccess">
    <slot />
  </div>
  <div v-else-if="showFallback" class="permission-denied">
    <div class="text-center py-8">
      <svg class="mx-auto h-12 w-12 text-gray-400" fill="none" viewBox="0 0 24 24" stroke="currentColor">
        <path stroke-linecap="round" stroke-linejoin="round" stroke-width="2" d="M12 15v2m-6 4h12a2 2 0 002-2v-6a2 2 0 00-2-2H6a2 2 0 00-2 2v6a2 2 0 002 2zm10-10V7a4 4 0 00-8 0v4h8z" />
      </svg>
      <h3 class="mt-2 text-sm font-medium text-gray-900">Sin permisos</h3>
      <p class="mt-1 text-sm text-gray-500">No tienes permisos para acceder a esta sección.</p>
    </div>
  </div>
</template>

<script setup lang="ts">
import { computed } from 'vue';
import { useRBAC } from '@/composables/useRBAC';

interface Props {
  permission?: string;
  permissions?: string[];
  role?: string;
  roles?: string[];
  requireAll?: boolean;
  showFallback?: boolean;
}

const props = withDefaults(defineProps<Props>(), {
  requireAll: false,
  showFallback: true
});

const { hasPermission, hasAnyPermission, hasAllPermissions, hasRole, hasAnyRole } = useRBAC();

const hasAccess = computed(() => {
  // Verificar permisos
  if (props.permission) {
    return hasPermission(props.permission);
  }
  
  if (props.permissions) {
    return props.requireAll 
      ? hasAllPermissions(props.permissions)
      : hasAnyPermission(props.permissions);
  }
  
  // Verificar roles
  if (props.role) {
    return hasRole(props.role);
  }
  
  if (props.roles) {
    return hasAnyRole(props.roles);
  }
  
  return false;
});
</script>

<style scoped>
.permission-denied {
  @apply bg-gray-50 rounded-lg border-2 border-dashed border-gray-300;
}
</style>
```

## 📊 Fase 7: Migración de Datos

### 7.1 Script de Migración

```sql
-- Migrar datos existentes de la tabla rol a la nueva estructura
INSERT INTO public.user_roles (user_id, role, restaurante_id)
SELECT 
    ui.id,
    CASE 
        WHEN r.rol ILIKE '%admin%' THEN 'restaurante_admin'::public.app_role
        WHEN r.rol ILIKE '%mesero%' THEN 'mesero'::public.app_role
        WHEN r.rol ILIKE '%cocina%' THEN 'cocina'::public.app_role
        WHEN r.rol ILIKE '%caja%' THEN 'caja'::public.app_role
        WHEN r.rol ILIKE '%inventario%' THEN 'inventario'::public.app_role
        ELSE 'cliente'::public.app_role
    END,
    ui.restaurante_id
FROM public.usuarios_info ui
LEFT JOIN public.rol r ON ui.rol_id = r.id
WHERE ui.id IS NOT NULL;

-- Verificar migración
SELECT 
    ur.role,
    COUNT(*) as user_count
FROM public.user_roles ur
GROUP BY ur.role
ORDER BY ur.role;
```

### 7.2 Script de Validación

```sql
-- Validar que todos los usuarios tengan roles asignados
SELECT 
    ui.id,
    ui.nombre_usuario,
    ur.role,
    ur.restaurante_id
FROM public.usuarios_info ui
LEFT JOIN public.user_roles ur ON ui.id = ur.user_id
WHERE ur.user_id IS NULL;

-- Verificar permisos por rol
SELECT 
    rp.role,
    COUNT(rp.permission) as permission_count,
    ARRAY_AGG(rp.permission ORDER BY rp.permission) as permissions
FROM public.role_permissions rp
GROUP BY rp.role
ORDER BY rp.role;
```
